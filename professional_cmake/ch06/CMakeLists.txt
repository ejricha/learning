cmake_minimum_required(VERSION 3.15)

# Set the project name
project(ch06)
message("\n\n[Project ${PROJECT_NAME}]\n")

# Good examples of common if statements
message("Check for MSVC (${MSVC}) instead of WIN32 (${WIN32})")
if (MSVC)
	message("  Compiling for MSVC")
else()
	message("  Not compiling for MSVC")
endif()
message("Check if CMAKE_GENERATOR (${CMAKE_GENERATOR}) is Xcode instead of APPLE (${APPLE})")
if (CMAKE_GENERATOR STREQUAL "Xcode")
	message("  Compiling for Xcode")
else()
	message("  Not compiling for Xcode")
endif()

# For loop with nested if statements
foreach(i RANGE 0 4)
	if(${i} LESS 1)
		message("i (${i}) is less than one")
	elseif(${i} EQUAL 1)
		message("i (${i}) is one")
	elseif(${i} LESS_EQUAL 2)
		message("i (${i}) is between one and two")
	else()
		message("i (${i}) is greater than two")
	endif()
endforeach()

# Set version lists
set(versionA 1.2    1.2    1.2.3  2.0.1  1.8.2)
set(versionB 1.2.0  1.2.3  1.2    1.9.7  2   )
list(LENGTH versionA listLength)
math(EXPR lastIndex "${listLength} - 1")
foreach(i RANGE 0 ${lastIndex})
	list(GET versionA ${i} a)
	list(GET versionB ${i} b)
	message("Comparing lists[${i}]")
	if(${a} VERSION_LESS ${b})
		message("  ${a} VERSION_LESS ${b}")
	endif()
	if(${a} VERSION_LESS_EQUAL ${b})
		message("  ${a} VERSION_LESS_EQUAL ${b}")
	endif()
	if(${a} VERSION_EQUAL ${b})
		message("  ${a} VERSION_EQUAL ${b}")
	endif()
	if(NOT ${a} VERSION_EQUAL ${b})
		message("  NOT ${a} VERSION_EQUAL ${b}")
	endif()
	if(${a} VERSION_GREATER_EQUAL ${b})
		message("  ${a} VERSION_GREATER_EQUAL ${b}")
	endif()
	if(${a} VERSION_GREATER ${b})
		message("  ${a} VERSION_GREATER ${b}")
	endif()
endforeach()
